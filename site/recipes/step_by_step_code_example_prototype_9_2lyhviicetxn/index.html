Details

  • The goal is to produce a step by step example to use with Rust playgrounds on typing-rust.alanwsmith.com
  • This one uses the ace editor.

const updateContent = (setIndex) => {
  s.setIndex = setIndex
  parts = []
  for (let i = 0; i <= s.setIndex; i++) {
    parts.push(s.lines[i])
  }
  s.editor.setValue(parts.join('\n'), 1)
}

const init = () => {
  loadLines()
  s.editor = ace.edit('editor')
  s.editor.setTheme('ace/theme/monokai')
  s.editor.session.setMode('ace/mode/rust')
  // updateContent(1)
  removeHighlights()
}

document.addEventListener('DOMContentLoaded', init)
